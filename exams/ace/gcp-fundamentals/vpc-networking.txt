# create firewall
gcloud compute --project=qwiklabs-gcp-02-3512e43afb81 firewall-rules create managementnet-allow-icmp-ssh-rdp \
   --direction=INGRESS \
	 --priority=1000 \
	 --network=managementnet \
	 --action=ALLOW \
	 --rules=icmp,tcp:22,tcp:3389 \
	 --source-ranges=0.0.0.0/0
